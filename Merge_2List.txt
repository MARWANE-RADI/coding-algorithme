def merge_2list(a,b):
	n = len(a)
	m = len(b)
	i = 0
	j = 0
	result = []
	while i < n and j < m:
	   if a[i] < b[j] :
		   result.append(a[i])
		   i += 1
	   else:
		   result.append(b[j])
		   j += 1
	while i < n:
		result.append(a[i])
		i += 1
	while j < m:
		result.append(b[j])
		i += 1
		
	return  result

a = [1,10,15,20]
b = [2,5,7,9]
New_Array = merge_2list(a,b)
print(New_Array)
